# E-1
## Solution
```bash
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(!head) return NULL;
        ListNode* current = head;
        while(current && current->next){
            if(current->val == current->next->val){
                ListNode* temp = current->next;
                current->next = temp->next;
            } else{
                current = current->next;
            }
        }
        return head;
    }
};
```
